    <script>
        // Sample module data
        let modules = [
            { id: "L-701", signal: "98%", location: "Zone A", condition: "Optimal" },
            { id: "L-702", signal: "75%", location: "Zone B", condition: "Needs Maintenance" },
            { id: "L-703", signal: "45%", location: "Zone C", condition: "Critical" }
        ];
        
        // Display modules
        function displayModules() {
            let html = '';
            modules.forEach(m => {
                let statusClass = m.condition.toLowerCase().includes('optimal') ? 'optimal' : 
                                 m.condition.toLowerCase().includes('maintenance') ? 'maintenance' : 'critical';
                
                html += `
                    <div class="card">
                        <h3>${m.id}</h3>
                        <p>📍 ${m.location}</p>
                        <p>📶 ${m.signal}</p>
                        <p class="${statusClass}">${m.condition}</p>
                        <button onclick="alert('Module ${m.id} verified!')">✓ Verify</button>
                    </div>
                `;
            });
            document.getElementById('modules').innerHTML = html;
        }
        
        // Simulate AI analysis
        function analyze() {
            let file = document.getElementById('fileInput').files[0];
            if(file) {
                alert("AI Processing Complete!\nModule: L-704\nSignal: 92%\nLocation: Zone D\nCondition: Optimal");
                
                // Add new module
                modules.push({
                    id: "L-704",
                    signal: "92%",
                    location: "Zone D",
                    condition: "Optimal"
                });
                
                // Update stats
                document.getElementById('total').innerText = modules.length;
                document.getElementById('optimal').innerText = '2';
                
                displayModules();
            } else {
                alert("Please select a file first!");
            }
        }
        
        // Initial display
        displayModules();
    </script>
